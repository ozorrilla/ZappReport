            //incluimos la librería
            require_once '../ZappReport.php';

            //creamos un objeto
            $report = new ZappReport();

            //indicamos la dirección de los reportes jrxml
            $report->setPath('jrxml/');

            //si exiten parametros los pasamos
            $parameters = array(
                'organismo' => 'ORMGE',
                'empresa' => 'CARTON',
                'finicio' => '01/07/2015',
                'ffin' => '01/08/2015'
            );

             /**
             * Cargamos la configuración del reporte.
             *
             * @param string $name nombre de fichero
             * @param array $data datos a mostrar. El formato debe ser el siguiente:
                          $data =  array(
                                    array('nombre'=>'Carlos', 'edad'=> 25),
                                    array('nombre'=>'Juan', 'edad'=> 17),
                                    array('nombre'=>'Maria', 'edad'=> 28),
                                    ...
                                );
             * @param array $parameters [opcional] parámetros del reporte.
             * @return El objeto ZappReport
             */
            $report->load('reportname', $data, $parameters);

            //lanzamos el reporte
            $report->show();
